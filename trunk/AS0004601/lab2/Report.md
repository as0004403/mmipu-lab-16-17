>PID controller formulas:

 u(k) = u(k-1) + Δu(k)

 Δu(k) = q0*e(k) + q1*e(k-1) + q2*e(k-2)

 q0 = K*(1+(Td/To))

 g1 = -K*(1+2*(Td/To) - (Td/To))

 g2 = K*(Td/To)

 e = uin-uprev

		double u_in;

		double e( int time)  override { 
			switch ( time)

				case 0:

			return q0 * e(0) + q1 * e(1) + q2 * e(2);
		}
